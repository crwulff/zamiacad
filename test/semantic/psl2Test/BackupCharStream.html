79      private final void FillBuff () throws java.io.IOException
80      {
81          // Buffer fill logic:
82          // If there is at least 2K left in this buffer, just read some more
83          // Otherwise, if we're processing a token and it either
84          // (a) starts in the other buffer (meaning it's filled both part of
85          //     the other buffer and some of this one, or
86          // (b) starts in the first 2K of this buffer (meaning its taken up
87          //     most of this buffer
88          // we expand this buffer.  Otherwise, we swap buffers.
89          // This guarantees we will be able to back up at least 2K characters.
90          if (curBuf.size - curBuf.dataLen < 2048)
91          {
92              if (tokenBeginPos >= 0
93                      && ((tokenBeginBuf == curBuf && tokenBeginPos < 2048)
94                      || tokenBeginBuf != curBuf))
95              {
96                  curBuf.expand(2048);
97              }
98              else
99              {
100                 swapBuf();
101                 curBuf.curPos = curBuf.dataLen = 0;
102             }
103         }
104 
105         try
106         {
107             int i = inputStream.read(curBuf.buffer, curBuf.dataLen,
108                     curBuf.size - curBuf.dataLen);
109             if (i == -1)
110             {
111                 inputStream.close();
112                 inputStreamClosed = true;
113                 throw new java.io.IOException();
114             }
115             else
116                 curBuf.dataLen += i;
117             return;
118         }
119         catch (java.io.IOException e)
120         {
121             if (curBuf.curPos > 0)
122                 --curBuf.curPos;
123             if (tokenBeginPos == -1)
124             {
125                 tokenBeginPos = curBuf.curPos;
126                 tokenBeginBuf = curBuf;
127             }
128             if (e.getClass().getName().equals("sun.io.MalformedInputException"))
129             {
130                 // it's an ugly hack, but we want to pass this exception
131                 // through the JavaCC parser, since it has a bad
132                 // exception handling
133                 throw new ParserRuntimeException("MalformedInput", e);
134             }
135             throw e;
136         }
137     }

149     private final void UpdateLineColumn (char c)
150     {
151         column++;
152 
153         if (prevCharIsLF)
154         {
155             prevCharIsLF = false;
156             line += (column = 1);
157         }
158         else if (prevCharIsCR)
159         {
160             prevCharIsCR = false;
161             if (c == '\n')
162             {
163                 prevCharIsLF = true;
164             }
165             else
166                 line += (column = 1);
167         }
168 
169         switch (c)
170         {
171             case '\r':
172                 prevCharIsCR = true;
173                 break;
174             case '\n':
175                 prevCharIsLF = true;
176                 break;
177             case '\t':
178                 column--;
179                 column += (8 - (column & 07));
180                 break;
181             default :
182                 break;
183         }
184 
185         curBuf.bufline[curBuf.curPos] = line;
186         curBuf.bufcolumn[curBuf.curPos] = column;
187     }

189     public final char readChar () throws java.io.IOException
190     {
191         // When we hit the end of the buffer, if we're backing up, we just
192         // swap, if we're not, we fill.
193         if (++curBuf.curPos >= curBuf.dataLen)
194         {
195             if (backupChars > 0)
196             {
197                 --curBuf.curPos;
198                 swapBuf();
199             }
200             else
201                 FillBuff();
202         }
203         ;
204 
205         // Don't mask off the high byte
206         char c = curBuf.buffer[curBuf.curPos];
207 
208         // No need to update line numbers if we've already processed this char
209         if (backupChars > 0)
210             --backupChars;
211         else
212             UpdateLineColumn(c);
213 
214         return (c);
215     }

257     public final void backup (int amount)
258     {
259         backupChars += amount;
260         if (curBuf.curPos - amount < 0)
261         {
262             int addlChars = amount - (inputStreamClosed ? 0 : 1) - curBuf.curPos;
263             curBuf.curPos = 0;
264             swapBuf();
265             curBuf.curPos = curBuf.dataLen - addlChars - 1;
266         }
267         else
268         {
269             curBuf.curPos -= amount;
270         }
271     }
